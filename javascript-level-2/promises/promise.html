The then() Method

// promise format // let myPromise = new Promise(function(resolve, reject) { // // do something // if (/* success */) { // resolve("Success message"); // } else { // reject("Error message"); // } // }); // myPromise.then( // function(value) { /* handle success */ }, // function(error) { /* handle error */ } // ); // promise.then(successHandler, errorHandler); //successHandler runs if the promise is resolved. //errorHandler runs if the promise is rejected.
